Slider demo #view docs

例子1

创建一个横向滑动条，设置其初始值为 50。

50

var slider = new $.ui.Slider( '#slider', {
    value : 50
});

var demoText = $( '#demoText1' );

slider.on( 'slide', function( e ){
    demoText.text( this.value );
});
